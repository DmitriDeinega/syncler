 151: /**
 152:  * Visible only for unit testing. Resolves a `$.foo.bar` JSONPath against the
 153:  * decrypted payload. Returns:
 154:  *  - the string value when the leaf is a JSON string,
 155:  *  - a JSON-encoded compact form when the leaf is a number/bool,
 156:  *  - null if any segment is missing or the leaf is an object/array/null.
 157:  *
 158:  * The publish-time validator enforces the `$.foo(.bar)*` syntax, but this
 159:  * function still tolerates a malformed input (returns null) rather than
 160:  * throwing ג€” defense against a future server change that loosens validation.
 161:  */
 162: internal fun resolveJsonPath(root: JSONObject, path: String): String? {
 163:     if (!path.startsWith("$.")) return null
 164:     val segments = path.substring(2).split('.')
 165:     if (segments.isEmpty() || segments.any { it.isEmpty() }) return null
 166: 
 167:     var current: Any? = root
 168:     for (segment in segments) {
 169:         val obj = current as? JSONObject ?: return null
 170:         if (!obj.has(segment)) return null
 171:         current = obj.opt(segment)
 172:         if (current == null || current == JSONObject.NULL) return null
 173:     }
 174:     return when (current) {
 175:         is String -> current
 176:         is Number, is Boolean -> current.toString()
 177:         else -> null
 178:     }
 179: }
